"""
Full Detect GUI Module
This script is launched from the main diff viewer GUI.
It receives a diff threshold and processes the same frame stack to detect and sort events.
"""

import os
import sys
import cv2
import numpy as np
import tkinter as tk
from tkinter import filedialog, Label, Button
from glob import glob
from collections import defaultdict

# --- CONFIGURATION ---
if len(sys.argv) >= 3:
    DIFF_THRESHOLD = int(sys.argv[1])
    STACK_FOLDER = sys.argv[2]
else:
    DIFF_THRESHOLD = 25
    STACK_FOLDER = "T:/Dahua_MKV_Streaker/Stacks/2025-06-15_00-00-00"

MASK_PATH = "T:/Dahua_MKV_Streaker/mask.png"

# --- HELPERS ---
def load_mask(path):
    if os.path.exists(path):
        mask = cv2.imread(path, cv2.IMREAD_GRAYSCALE)
        return cv2.threshold(mask, 127, 255, cv2.THRESH_BINARY)[1]
    return None

def detect_events(diff_img, min_area=40):
    contours, _ = cv2.findContours(diff_img, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
    detections = []

    for cnt in contours:
        area = cv2.contourArea(cnt)
        if area < min_area:
            continue

        perimeter = cv2.arcLength(cnt, True)
        if perimeter == 0:
            continue

        circularity = 4 * np.pi * area / (perimeter ** 2)
        x, y, w, h = cv2.boundingRect(cnt)
        aspect_ratio = w / h if h != 0 else 0

        classification = "flare" if circularity > 0.6 and aspect_ratio < 1.5 else "streaker"

        detections.append({
            "bbox": (x, y, w, h),
            "area": area,
            "aspect": aspect_ratio,
            "circ": circularity,
            "type": classification,
            "centroid": (x + w // 2, y + h // 2)
        })

    return detections

def group_detections_by_time(detection_log, max_gap=2):
    """
    Groups frames with detections into temporal groups where
    frames are no more than `max_gap` frames apart.
    """
    keys = sorted(detection_log.keys())
    groups = []
    current_group = []

    for i, k in enumerate(keys):
        if not current_group:
            current_group.append(k)
        elif k - keys[i - 1] <= max_gap:
            current_group.append(k)
        else:
            groups.append(list(current_group))
            current_group = [k]

    if current_group:
        groups.append(current_group)

    return groups

# --- GUI SETUP ---
class DetectionGUI:
    def __init__(self, master):
        self.master = master
        master.title("Full Detection Review")

        self.label = Label(master, text=f"Running detection with threshold: {DIFF_THRESHOLD}\nStack folder: {STACK_FOLDER}")
        self.label.pack()

        self.run_button = Button(master, text="Run Detection", command=self.run_detection)
        self.run_button.pack()

    def run_detection(self):
        mask = load_mask(MASK_PATH)
        detection_log = defaultdict(list)

        frame_paths = sorted(glob(os.path.join(STACK_FOLDER, "*.png")))
        for i, img_path in enumerate(frame_paths):
            frame = cv2.imread(img_path, cv2.IMREAD_GRAYSCALE)
            if frame is None:
                continue

            if mask is not None:
                frame = cv2.bitwise_and(frame, frame, mask=mask)

            diff = cv2.threshold(frame, DIFF_THRESHOLD, 255, cv2.THRESH_BINARY)[1]
            detections = detect_events(diff)

            if detections:
                detection_log[i].extend(detections)

        flare_count = sum(1 for v in detection_log.values() for d in v if d["type"] == "flare")
        streak_count = sum(1 for v in detection_log.values() for d in v if d["type"] == "streaker")

        time_groups = group_detections_by_time(detection_log)
        result_text = (
            f"Flares: {flare_count}\n"
            f"Streakers: {streak_count}\n"
            f"Detection Groups: {len(time_groups)}"
        )
        self.label.config(text=result_text)

# --- MAIN EXECUTION ---
if __name__ == "__main__":
    root = tk.Tk()
    gui = DetectionGUI(root)
    root.mainloop()
